	1. Check PDB users
		SELECT username, account_status, default_tablespace, temporary_tablespace
			FROM dba_users;
			
	2. Roles, grants, and privileges
		-- Roles
		i) SELECT * FROM dba_roles;

		-- Role grants
		ii) SELECT * FROM dba_role_privs;

		-- System privileges
		iii) SELECT * FROM dba_sys_privs;

		-- Object privileges
		iv) SELECT * FROM dba_tab_privs;
		
	3. Tables, indexes, views, sequences
		-- Tables and row count
		i) SELECT table_name, num_rows FROM user_tables;

		-- Indexes
		ii) SELECT index_name, table_name, uniqueness FROM user_indexes;

		-- Views
		iii) SELECT view_name FROM user_views;

		-- Sequences
		iv) SELECT sequence_name, last_number FROM user_sequences;
		
	4. Triggers, constraints
		-- Constraints
		i) SELECT constraint_name, table_name, constraint_type FROM user_constraints;

		-- Triggers
		ii) SELECT trigger_name, table_name, status FROM user_triggers;
		
	5. Tablespaces and storage (PDB-specific)
		i) SELECT tablespace_name, status, contents, extent_management FROM dba_tablespaces;

		ii) SELECT file_name, bytes/1024/1024 AS size_mb, autoextensible FROM dba_data_files;
		
	6. Jobs, scheduler, and alerts
		i) SELECT job_name, enabled, state, last_start_date FROM dba_scheduler_jobs;
		
	7. Database links in PDB
		i) SELECT db_link, username, host FROM dba_db_links;
	8. Performance / session info
		i) SELECT sid, serial#, username, status, osuser, machine FROM v$session;
		
*** Optional: Export a full snapshot of PDB info ***
		-- Object summary
		i) SELECT object_type, COUNT(*) FROM user_objects GROUP BY object_type;

		-- Row counts per table
		ii) SELECT table_name, num_rows FROM user_tables ORDER BY table_name;
